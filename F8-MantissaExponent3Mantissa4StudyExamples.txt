0 0000 001 = 0.125 * 2^(1-7)
0 0000 010 = 0.250 * 2^(1-7)
0 0000 011 = 0.375 * 2^(1-7)
0 0000 100 = 0.500 * 2^(1-7)
0 0000 101 = 0.625 * 2^(1-7)
0 0000 110 = 0.750 * 2^(1-7)
0 0000 111 = 0.875 * 2^(1-7)

0 0001 000 = 1.000 * 2^(1-7)
0 0001 001 = 1.125 * 2^(1-7)
0 0001 010 = 1.250 * 2^(1-7)
0 0001 011 = 1.375 * 2^(1-7)
0 0001 100 = 1.500 * 2^(1-7)
0 0001 101 = 1.625 * 2^(1-7)
0 0001 110 = 1.750 * 2^(1-7)
0 0001 111 = 1.875 * 2^(1-7)


0 0111 000 = 1.000 * 2^(7-7)
0 0111 001 = 1.125 * 2^(7-7)


0 1110 000 = 1.000 * 2^(14-7) = 128
0 1110 001 = 1.125 * 2^(14-7) = 144

0 1110 111 = 1.875 * 2^(14-7) = 240 (Greatest normalized number)
